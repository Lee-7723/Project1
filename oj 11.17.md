>   \#include\<math.h\>

>   main()

>   { float a,b,c,p,s;

>   char x;

>   scanf("%f%f%f",\&a,&b,&c);

>   if(a+b\>c&&a+c\>b&&b+c\>a) //判断是否可以构成三角形

>   { p=(a+b+c)/2;

>   s=sqrt(p\*(p-a)\*(p-b)\*(p-c));

>   if(a==b&&a==c&&b==c) //等边

>   x='E';

>   if((a==b&&a!=c)\|\|(a==c&&a!=b)\|\|(b=c&&b!=a)) //等腰

>   x='I';

>   if(a!=b&&a!=c&&b!=c) //普通

>   x='O';

>   printf("%c%.2f",x,s);

>   }

>   }
