{

scanf("%d%d",\&a,&b);

for(i=a;i\<=b;i++)

{

if(hui(i)&&zhi(i)) printf("%d\\n",i);

}

}

int hui(int i)

{

int h,n;

n=i;

h=0;

while(n)

{ h=h\*10+n%10;

n=n/10; }

return (h==i);

} //求其倒序数

int zhi(int i)

{ int j,k;

k=sqrt(i);

for(j=2;j\<=k;j++)

{if(i%j==0) return 0;} //若非质数，返回值为0

return 1; }
