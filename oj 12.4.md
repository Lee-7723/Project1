4. 求矩阵行列乘积
=================

![](media/ab5c9f25014009f3feb7382a6ec92fdc.png)

\#include\<stdio.h\>

main()

{ int n,i,j,ji;

scanf("%d",\&n);

int a[n+1][n+1];

for(i=0;i\<n;i++)

{ for(j=0;j\<n;j++)

{ scanf("%d",\&a[i][j]);}

} //输入原始矩阵

for(i=0;i\<n;i++)

{ for(j=0,ji=1;j\<n;j++)

{ ji\*=a[i][j];}

a[i][n]=ji;} //给最后一个赋值，即行乘积

for(j=0;j\<n;j++)

{ for(i=0,ji=1;i\<n;i++)

{ ji\*=a[i][j];}

a[n][j]=ji;} //给最后一个赋值，即列乘积

for(i=0;i\<n;i++)

{ for(j=0;j\<n+1;j++)

{ printf("%d",a[i][j]);

j!=n?printf(" "): printf("\\n")；} } //输出原矩阵与行的乘积，注意控制空格与换行

for(j=0;j\<n;j++)

{ printf("%d",a[n][j]);

if(j\<n-1)

printf(" ");} //输出列的乘积

}
